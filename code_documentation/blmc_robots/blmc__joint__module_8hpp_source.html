661  GoToReturnCode go_to(Vector angle_to_reach_rad,

664  // Compute a min jerk trajectory
665  Vector initial_joint_positions = get_measured_angles();

667  .array().abs().maxCoeff() /

676  }

679  real_time_tools::Spinner spinner;
680  double sampling_period = 0.001; // TODO magic number
681  spinner.set_period(sampling_period);
682  GoToReturnCode go_to_status;

684  do{

689  double desired_torque =

691  modules_[i]->set_torque(desired_torque);
692  modules_[i]->send_torque();
693  }
694  go_to_status = GoToReturnCode::RUNNING;

705  }
