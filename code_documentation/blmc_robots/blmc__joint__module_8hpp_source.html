552  Vector home_offset_rad,

554  {
555  // Initialise homing for all joints
556  for(size_t i = 0; i < COUNT; i++)
557  {
558  modules_[i]->init_homing((int) i, search_distance_limit_rad,
559  home_offset_rad[i], profile_step_size_rad);
560  }
561 
562  // run homing for all joints until all of them are done
563  real_time_tools::Spinner spinner;
564  spinner.set_period(0.001); // TODO magic number
565  HomingReturnCode homing_status;
566  do {
567  bool all_succeeded = true;
568  homing_status = HomingReturnCode::RUNNING;
569 
570  for(size_t i = 0; i < COUNT; i++)
571  {
572  HomingReturnCode joint_result = modules_[i]->update_homing();
573 
574  all_succeeded &= (joint_result == HomingReturnCode::SUCCEEDED);
575 
576  if (joint_result == HomingReturnCode::NOT_INITIALIZED ||
577  joint_result == HomingReturnCode::FAILED) {
578  homing_status = joint_result;
579  // abort homing
580  break;
581  }
582  }
583 
584  if (all_succeeded) {
585  homing_status = HomingReturnCode::SUCCEEDED;
586  }
587 
588  spinner.spin();
589  } while (homing_status == HomingReturnCode::RUNNING);
590 
591  return homing_status;
592  }
