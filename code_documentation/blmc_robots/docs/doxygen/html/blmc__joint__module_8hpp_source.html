615  HomingReturnCode execute_homing(
616  double search_distance_limit_rad,
617  Vector home_offset_rad,
618  Vector profile_step_size_rad = Vector::Constant(0.001))
619  {
620  // Initialise homing for all joints
621  for (size_t i = 0; i < COUNT; i++)
622  {
623  modules_[i]->init_homing((int)i,
624  search_distance_limit_rad,
625  home_offset_rad[i],
626  profile_step_size_rad[i]);
627  }
628 
629  // run homing for all joints until all of them are done
630  real_time_tools::Spinner spinner;
631  spinner.set_period(0.001); // TODO magic number
632  HomingReturnCode homing_status;
633  do
634  {
635  bool all_succeeded = true;
636  homing_status = HomingReturnCode::RUNNING;
637 
638  for (size_t i = 0; i < COUNT; i++)
639  {
640  HomingReturnCode joint_result = modules_[i]->update_homing();
641 
642  all_succeeded &= (joint_result == HomingReturnCode::SUCCEEDED);
643 
644  if (joint_result == HomingReturnCode::NOT_INITIALIZED ||
645  joint_result == HomingReturnCode::FAILED)
646  {
647  homing_status = joint_result;
648  // abort homing
649  break;
650  }
651  }
652  if (homing_status == HomingReturnCode::RUNNING)
653  {
654  for (unsigned i = 0; i < COUNT; ++i)
655  {
656  modules_[i]->send_torque();
657  }
658  }
659 
660  if (all_succeeded)
661  {
662  homing_status = HomingReturnCode::SUCCEEDED;
663  }
664 
665  spinner.spin();
666  } while (homing_status == HomingReturnCode::RUNNING);
667 
668  return homing_status;
669  }

679  GoToReturnCode go_to(Vector angle_to_reach_rad,
680  double average_speed_rad_per_sec = 1.0)
681  {
682  // Compute a min jerk trajectory
683  Vector initial_joint_positions = get_measured_angles();
684  double final_time = (angle_to_reach_rad - initial_joint_positions)
685  .array()
686  .abs()
687  .maxCoeff() /
688  average_speed_rad_per_sec;
689 
690  std::array<TimePolynome<5>, COUNT> min_jerk_trajs;
691  for (unsigned i = 0; i < COUNT; i++)
692  {
693  min_jerk_trajs[i].set_parameters(final_time,
694  initial_joint_positions[i],
695  0.0 /*initial speed*/,
696  angle_to_reach_rad[i]);
697  }
698 
699  // run got_to for all joints
700  real_time_tools::Spinner spinner;
701  double sampling_period = 0.001; // TODO magic number
702  spinner.set_period(sampling_period);
703  GoToReturnCode go_to_status;
704  double current_time = 0.0;
705  do
706  {
707  // TODO: add a security if error gets too big
708  for (unsigned i = 0; i < COUNT; i++)
709  {
710  double desired_pose = min_jerk_trajs[i].compute(current_time);
711  double desired_torque =
712  modules_[i]->execute_position_controller(desired_pose);
713  modules_[i]->set_torque(desired_torque);
714  }
715  for (unsigned i = 0; i < COUNT; ++i)
716  {
717  modules_[i]->send_torque();
718  }
719  go_to_status = GoToReturnCode::RUNNING;
720 
721  current_time += sampling_period;
722  spinner.spin();
723 
724  } while (current_time < (final_time + sampling_period));
725 
726  // Stop all motors (0 torques) after the destination achieved
727  for (unsigned i = 0; i < COUNT; i++)
728  {
729  modules_[i]->set_torque(0.0);
730  }
731  for (unsigned i = 0; i < COUNT; ++i)
732  {
733  modules_[i]->send_torque();
734  }
735 
736  Vector final_pos = get_measured_angles();
737  if ((angle_to_reach_rad - final_pos).isMuchSmallerThan(1.0, 1e-3))
738  {
739  go_to_status = GoToReturnCode::SUCCEEDED;
740  }
741  else
742  {
743  go_to_status = GoToReturnCode::FAILED;
744  }
745  return go_to_status;
746  }
