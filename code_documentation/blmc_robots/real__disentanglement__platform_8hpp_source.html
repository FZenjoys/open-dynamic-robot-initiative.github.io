183  void calibrate()
184  {
188  real_time_tools::Spinner spinner;
189  spinner.set_period(0.001);
190  std::vector<Vector> running_velocities(1000);
191  int running_index = 0;
192  Vector sum = Vector::Zero();
193  while(running_index < 3000 || (sum.maxCoeff() / 1000.0 > 0.001))
194  {
195  Vector torques = -1 * get_max_torques();
196  torques(table) = 0; // we do not calibrate the table since it has no stop
197  torques(tip) = -torques(tip);
198 
199  constrain_and_apply_torques(torques);
200  Vector velocities = get_measured_velocities();
201  if (running_index >= 1000)
202  sum = sum - running_velocities[running_index % 1000];
203  running_velocities[running_index % 1000] = velocities;
204  sum = sum + velocities;
205  running_index++;
206  spinner.spin();
207  }
208  int count = 0;
209  int linearly_decrease_time_steps = 1000;
210  int zero_torque_time_steps = 500;
211  while(count < linearly_decrease_time_steps)
212  {
213  Vector torques = ((linearly_decrease_time_steps -
214  count + 0.0) / linearly_decrease_time_steps) *
215  get_max_torques() * -1;
216  torques(table) = 0; // we do not calibrate the table since it has no stop
217  torques(tip) = -torques(tip);
218 
219 
220  constrain_and_apply_torques(torques);
221  count++;
222  spinner.spin();
223  }
224  count = 0;
225  while(count < zero_torque_time_steps)
226  {
227  Vector torques = Vector::Zero();
228  constrain_and_apply_torques(torques);
229  count++;
230  spinner.spin();
231  }
232  Vector angle_offsets = get_measured_angles();
233  angle_offsets(table) = 0;
234 
235  Vector zero_angles_relative_to_joint_stops(0.0, 0.0, -0.506);
236  set_zero_angles(angle_offsets + zero_angles_relative_to_joint_stops);
237  }
