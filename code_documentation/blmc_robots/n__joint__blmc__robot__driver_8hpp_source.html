416  break;
417  case ErrorCodes::CRIT_TEMP:

419  break;
420  case ErrorCodes::POSCONV:
421  board_error_msg =

423  break;
424  case ErrorCodes::POS_ROLLOVER:
425  board_error_msg = "Position Rollover";

428  board_error_msg = "Other Error";

519  bool home_on_index_after_negative_end_stop(
520  double endstop_search_torque_Nm,
521  Vector home_offset_rad = Vector::Zero())
522  {
525  const double SEARCH_DISTANCE_LIMIT_RAD =
526  (1.5 / motor_parameters_.gear_ratio) * 2 * M_PI;
527 
528  rt_printf("Start homing.\n");
529  if (has_endstop_)
530  {
532  constexpr uint32_t MIN_STEPS_MOVE_TO_END_STOP = 1000;
534  constexpr uint32_t SIZE_VELOCITY_WINDOW = 100;
536  constexpr double STOP_VELOCITY = 0.01;
537 
538  static_assert(MIN_STEPS_MOVE_TO_END_STOP > SIZE_VELOCITY_WINDOW,
539  "MIN_STEPS_MOVE_TO_END_STOP has to be bigger than"
540  " SIZE_VELOCITY_WINDOW to ensure correct computation"
541  " of average velocity.");
542 
543  // Move until velocity drops to almost zero (= joints hit the end
544  // stops) but at least for MIN_STEPS_MOVE_TO_END_STOP time steps.
545  // TODO: add timeout to this loop?
546  std::vector<Vector> running_velocities(SIZE_VELOCITY_WINDOW);
547  Vector summed_velocities = Vector::Zero();
548  uint32_t step_count = 0;
549  while (step_count < MIN_STEPS_MOVE_TO_END_STOP ||
550  (summed_velocities.maxCoeff() / SIZE_VELOCITY_WINDOW >
551  STOP_VELOCITY))
552  {
553  Vector torques = Vector::Constant(-endstop_search_torque_Nm);
554  apply_action_uninitialized(torques);
555  Vector abs_velocities =
556  get_latest_observation().velocity.cwiseAbs();
557 
558  uint32_t running_index = step_count % SIZE_VELOCITY_WINDOW;
559  if (step_count >= SIZE_VELOCITY_WINDOW)
560  {
561  summed_velocities -= running_velocities[running_index];
562  }
563  running_velocities[running_index] = abs_velocities;
564  summed_velocities += abs_velocities;
565  step_count++;
566 
567 #ifdef VERBOSE
568  Eigen::IOFormat commainitfmt(
569  4, Eigen::DontAlignCols, " ", " ", "", "", "", "");
570  std::cout << ((summed_velocities / SIZE_VELOCITY_WINDOW)
571  .array() > STOP_VELOCITY)
572  .format(commainitfmt)
573  << std::endl;
574 #endif
575  }
576  rt_printf("Reached end stop.\n");
577  }
578 
579  // Home on encoder index
580  HomingReturnCode homing_status = joint_modules_.execute_homing(
581  SEARCH_DISTANCE_LIMIT_RAD, home_offset_rad);
582 
583  rt_printf("Finished homing.\n");
584 
585  return homing_status == HomingReturnCode::SUCCEEDED;
586  }

599  bool move_to_position(const Vector &goal_pos,
600  const double tolerance,
601  const uint32_t timeout_cycles)
602  {
603  bool reached_goal = false;
604  uint32_t cycle_count = 0;
605 
606  while (!reached_goal && cycle_count < timeout_cycles)
607  {
608  apply_action(Action::Position(goal_pos));
609 
610  const Vector position_error =
611  goal_pos - get_latest_observation().position;
612  const Vector velocity = get_latest_observation().velocity;
613 
614  // Check if the goal is reached (position error below tolerance and
615  // velocity close to zero).
616  constexpr double ZERO_VELOCITY = 1e-4;
617  reached_goal = ((position_error.array().abs() < tolerance).all() &&
618  (velocity.array().abs() < ZERO_VELOCITY).all());
619 
620  cycle_count++;
621  }
622 
623  return reached_goal;
624  }
