187  Action apply_action_uninitialized(const Action &desired_action)
188  {
189  double start_time_sec = real_time_tools::Timer::get_current_time_sec();
190 
191  Observation observation = get_latest_observation();
192 
193  Action applied_action;
194 
195  applied_action.torque = desired_action.torque;
196  applied_action.position = desired_action.position;
197 
198  // Position controller
199  // -------------------
200  // TODO: add position limits
201  //
202  // NOTE: in the following lines, the Eigen function `unaryExpr` is
203  // used to determine which elements of the vectors are NaN. This is
204  // because `isNaN()` was only added in Eigen 3.3 but we have to be
205  // compatible with 3.2.
206  // The std::isnan call needs to be wrapped in a lambda because it is
207  // overloaded and unaryExpr cannot resolve which version to use when
208  // passed directly.
209 
210  // Run the position controller only if a target position is set for at
211  // least one joint.
212  if (!applied_action.position
213  .unaryExpr([](double x) { return std::isnan(x); })
214  .all())
215  {
216  // Replace NaN-values with default gains
217  applied_action.position_kp =
218  applied_action.position_kp
219  .unaryExpr([](double x) { return std::isnan(x); })
220  .select(default_position_kp_, desired_action.position_kp);
221  applied_action.position_kd =
222  applied_action.position_kd
223  .unaryExpr([](double x) { return std::isnan(x); })
224  .select(default_position_kd_, desired_action.position_kd);
225 
226  Vector position_error =
227  applied_action.position - observation.position;
228 
229  // simple PD controller
230  Vector position_control_torque =
231  applied_action.position_kp.cwiseProduct(position_error) +
232  applied_action.position_kd.cwiseProduct(observation.velocity);
233 
234  // position_control_torque contains NaN for joints where target
235  // position is set to NaN! Filter those out and set the torque to
236  // zero instead.
237  position_control_torque =
238  position_control_torque
239  .unaryExpr([](double x) { return std::isnan(x); })
240  .select(0, position_control_torque);
241 
242  // Add result of position controller to the torque command
243  applied_action.torque += position_control_torque;
244  }
245 
246  // Safety Checks
247  // -------------
248  // limit to configured maximum torque
249  applied_action.torque =
250  mct::clamp(applied_action.torque, -max_torque_Nm_, max_torque_Nm_);
251  // velocity damping to prevent too fast movements
252  applied_action.torque -= safety_kd_.cwiseProduct(observation.velocity);
253  // after applying checks, make sure we are still below the max. torque
254  applied_action.torque =
255  mct::clamp(applied_action.torque, -max_torque_Nm_, max_torque_Nm_);
256 
257  joint_modules_.set_torques(applied_action.torque);
258  joint_modules_.send_torques();
259 
260  real_time_tools::Timer::sleep_until_sec(start_time_sec + 0.001);
261 
262  return applied_action;
263  }

296  bool home_on_index_after_negative_end_stop(
297  double torque_ratio, Vector home_offset_rad = Vector::Zero())
298  {
300  constexpr double SEARCH_DISTANCE_LIMIT_RAD = 2.0;
301  // TODO distance limit could be set based on gear ratio to be 1.5 motor
302  // revolutions
303 
304  if (has_endstop_)
305  {
307  constexpr uint32_t MIN_STEPS_MOVE_TO_END_STOP = 1000;
309  constexpr uint32_t SIZE_VELOCITY_WINDOW = 100;
311  constexpr double STOP_VELOCITY = 0.001;
312 
313  static_assert(MIN_STEPS_MOVE_TO_END_STOP > SIZE_VELOCITY_WINDOW,
314  "MIN_STEPS_MOVE_TO_END_STOP has to be bigger than"
315  " SIZE_VELOCITY_WINDOW to ensure correct computation"
316  " of average velocity.");
317 
318  // Move until velocity drops to almost zero (= joints hit the end
319  // stops) but at least for MIN_STEPS_MOVE_TO_END_STOP time steps.
320  // TODO: add timeout to this loop?
321  std::vector<Vector> running_velocities(SIZE_VELOCITY_WINDOW);
322  Vector summed_velocities = Vector::Zero();
323  uint32_t step_count = 0;
324  while (step_count < MIN_STEPS_MOVE_TO_END_STOP ||
325  (summed_velocities.maxCoeff() / SIZE_VELOCITY_WINDOW >
326  STOP_VELOCITY))
327  {
328  Vector torques = -1 * torque_ratio * get_max_torques();
329  apply_action_uninitialized(torques);
330  Vector abs_velocities =
331  get_latest_observation().velocity.cwiseAbs();
332 
333  uint32_t running_index = step_count % SIZE_VELOCITY_WINDOW;
334  if (step_count >= SIZE_VELOCITY_WINDOW)
335  {
336  summed_velocities -= running_velocities[running_index];
337  }
338  running_velocities[running_index] = abs_velocities;
339  summed_velocities += abs_velocities;
340  step_count++;
341  }
342  }
343 
344  // Home on encoder index
345  HomingReturnCode homing_status = joint_modules_.execute_homing(
346  SEARCH_DISTANCE_LIMIT_RAD, home_offset_rad);
347 
348  rt_printf("Finished homing.\n");
349 
350  return homing_status == HomingReturnCode::SUCCEEDED;
351  }

364  bool move_to_position(const Vector &goal_pos,
365  const double tolerance,
366  const uint32_t timeout_cycles)
367  {
368  bool reached_goal = false;
369  uint32_t cycle_count = 0;
370 
371  while (!reached_goal && cycle_count < timeout_cycles)
372  {
373  apply_action(Action::Position(goal_pos));
374 
375  const Vector position_error =
376  goal_pos - get_latest_observation().position;
377  const Vector velocity = get_latest_observation().velocity;
378 
379  // Check if the goal is reached (position error below tolerance and
380  // velocity close to zero).
381  constexpr double ZERO_VELOCITY = 1e-4;
382  reached_goal = ((position_error.array().abs() < tolerance).all() &&
383  (velocity.array().abs() < ZERO_VELOCITY).all());
384 
385  cycle_count++;
386  }
387 
388  return reached_goal;
389  }
