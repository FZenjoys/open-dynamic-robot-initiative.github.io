150  bool home_on_index_after_negative_end_stop(
151  double torque_ratio, Vector home_offset_rad=Vector::Zero())
152  {
156 
158  constexpr uint32_t MIN_STEPS_MOVE_TO_END_STOP = 1000;
160  constexpr uint32_t SIZE_VELOCITY_WINDOW = 100;
162  constexpr double STOP_VELOCITY = 0.001;
164  constexpr double SEARCH_DISTANCE_LIMIT_RAD = 2.0;
165 
166  static_assert(MIN_STEPS_MOVE_TO_END_STOP > SIZE_VELOCITY_WINDOW,
167  "MIN_STEPS_MOVE_TO_END_STOP has to be bigger than"
168  " SIZE_VELOCITY_WINDOW to ensure correct computation"
169  " of average velocity.");
170 
171 
172  // Move until velocity drops to almost zero (= joints hit the end stops)
173  // but at least for MIN_STEPS_MOVE_TO_END_STOP time steps.
174  // TODO: add timeout to this loop?
175  std::vector<Vector> running_velocities(SIZE_VELOCITY_WINDOW);
176  Vector summed_velocities = Vector::Zero();
177  int step_count = 0;
178  while (step_count < MIN_STEPS_MOVE_TO_END_STOP
179  || (summed_velocities.maxCoeff() / SIZE_VELOCITY_WINDOW > STOP_VELOCITY))
180  {
181  Vector torques = -1 * torque_ratio * get_max_torques();
182  TimeIndex t = append_desired_action(torques);
183  Vector abs_velocities = get_observation(t).velocity.cwiseAbs();
184 
185  uint32_t running_index = step_count % SIZE_VELOCITY_WINDOW;
186  if (step_count >= SIZE_VELOCITY_WINDOW) {
187  summed_velocities -= running_velocities[running_index];
188  }
189  running_velocities[running_index] = abs_velocities;
190  summed_velocities += abs_velocities;
191  step_count++;
192  }
193 
194  // need to "pause" as the desired actions queue is not filled while
195  // homing is running.
196  pause();
197 
198  // Home on encoder index
199  HomingReturnCode homing_status = joint_modules_.execute_homing(
200  SEARCH_DISTANCE_LIMIT_RAD, home_offset_rad);
201 
202  rt_printf("Finished homing.\n");
203 
204  return homing_status == HomingReturnCode::SUCCEEDED;
205  }

221  bool move_to_position(const Vector &goal_pos, const double kp,
222  const double kd, const double tolerance,
223  const uint32_t timeout_cycles)
224  {
228 
229  bool reached_goal = false;
230  int cycle_count = 0;
231  Vector desired_torque = Vector::Zero();
232  Eigen::Vector3d last_diff(std::numeric_limits<double>::max(),
233  std::numeric_limits<double>::max(),
234  std::numeric_limits<double>::max());
235 
236  while (!reached_goal && cycle_count < timeout_cycles)
237  {
238  const TimeIndex t = append_desired_action(desired_torque);
239 
240  const Vector position_error = goal_pos - get_observation(t).angle;
241  const Vector velocity = get_observation(t).velocity;
242 
243  // we implement here a small PD control at the current level
244  desired_torque = kp * position_error - kd * velocity;
245 
246 #ifdef VERBOSE
247  if (cycle_count % 100 == 0) {
248  rt_printf("--\n"
249  "position error: %10.5f, %10.5f, %10.5f\n"
250  "velocity: %10.5f, %10.5f, %10.5f\n",
251  position_error[0], position_error[1], position_error[2],
252  velocity[0], velocity[1], velocity[2]);
253  }
254 #endif
255 
256  // Check if the goal is reached (position error below tolerance and
257  // velocity close to zero).
258  constexpr double ZERO_VELOCITY = 1e-4;
259  reached_goal = (
260  (position_error.array().abs() < tolerance).all() &&
261  (velocity.array().abs() < ZERO_VELOCITY).all()
262  );
263 
264  cycle_count++;
265  }
266 
267  return reached_goal;
268  }

281  void calibrate()

283  constexpr double TORQUE_RATIO = 0.6;
284  constexpr double CONTROL_GAIN_KP = 0.4;
285  constexpr double CONTROL_GAIN_KD = 0.0025;
286  constexpr double POSITION_TOLERANCE_RAD = 0.2;
287  constexpr double MOVE_TIMEOUT = 2000;
288 
289  // Offset between home position and zero. Defined such that the zero
290  // position is at the negative end stop (for compatibility with old
291  // homing method).
292  Vector home_offset_rad;
293  home_offset_rad << -0.17, -0.54, 0.0;
294 
295  // Start position to which the robot moves after homing.
296  Vector starting_position_rad;
297  starting_position_rad << 1.5, 1.5, 3.0;
298 
299  joint_modules_.set_position_control_gains(CONTROL_GAIN_KP,
300  CONTROL_GAIN_KD);
301 
302  bool is_homed = home_on_index_after_negative_end_stop(TORQUE_RATIO,
303  home_offset_rad);
304 
305  if (is_homed) {
306  bool reached_goal = move_to_position(starting_position_rad,
307  CONTROL_GAIN_KP,
308  CONTROL_GAIN_KD,
309  POSITION_TOLERANCE_RAD,
310  MOVE_TIMEOUT);
311  if (!reached_goal) {
312  rt_printf("Failed to reach goal, timeout exceeded.\n");
313  }
314  }
315 
316  pause();
317  }
