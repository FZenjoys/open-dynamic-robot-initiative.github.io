442  void loop()
443  {
444  timers_.resize(10);
445  real_time_tools::set_cpu_dma_latency(0);
446 
447  // wait until first desired_action was received
448  // ----------------------------
449  while (!destructor_was_called_ &&
450  !robot_data_->desired_action->wait_for_timeindex(0, 0.1))
451  {
452  }
453 
454  for (long int t = 0; !destructor_was_called_; t++)
455  {
456  // TODO: figure out latency stuff!! open /dev/cpu_dma_latency:
457  // Permission denied
458 
459  timers_[0].tac_tic();
460 
461  timers_[6].tic();
462  // get latest observation from robot and append it to robot_data_
463  // --------
464  Observation observation = robot_driver_.get_latest_observation();
465  timers_[6].tac();
466 
467  timers_[1].tic();
468  robot_data_->observation->append(observation);
469  // TODO: for some reason this sometimes takes more than 2 ms
470  // i think this may be due to a non-realtime thread blocking the
471  // timeseries. this is in fact an issue, we might have to duplicate
472  // all the timeseries and have a realtime thread writing back and
473  // forth
474  timers_[1].tac();
475 
476  timers_[2].tic();
477  // if the robot has a finite max_inter_action_duration_s (not NAN or
478  // infinite, meaning it requires receiving actions in fixed time
479  // intervals), but robot_data_ has not received yet the next action
480  // to apply, we optionally repeat the previous action.
481  Status status = {0};
482  if (std::isfinite(
483  robot_driver_.get_max_inter_action_duration_s()) &&
484  robot_data_->desired_action->newest_timeindex() < t)
485  {
486  uint32_t action_repetitions =
487  robot_data_->status->newest_element().action_repetitions;
488 
489  if (action_repetitions < max_action_repetitions_)
490  {
491  robot_data_->desired_action->append(
492  robot_data_->desired_action->newest_element());
493  status.action_repetitions = action_repetitions + 1;
494  }
495  }
496  robot_data_->status->append(status);
497  timers_[2].tac();
498 
499  timers_[3].tic();
500  // TODO: this may wait forever
501  Action desired_action = (*robot_data_->desired_action)[t];
502  timers_[3].tac();
503  timers_[4].tic();
504  Action applied_action = robot_driver_.apply_action(desired_action);
505  timers_[4].tac();
506  timers_[5].tic();
507  robot_data_->applied_action->append(applied_action);
508  timers_[5].tac();
509 
510  // if (t % 5000 == 0) {
