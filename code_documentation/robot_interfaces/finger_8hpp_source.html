152  void loop()
153  {
154  // TODO: there is a slight problem here: this thread
155  // may start running before child class is created, and
156  // hence attempt to call nonexistant function.
157  for (TimeIndex t = 0; true; t++)
158  {
159  if (is_realtime_ && is_paused_ &&
160  (desired_action_->length() == 0 ||
161  desired_action_->newest_timeindex() < t))
162  {
163  desired_action_->append(Action::Zero());
164  }
165 
166  Action desired_action = (*desired_action_)[t];
167  Observation observation = get_latest_observation();
168  Action safe_action = compute_safe_action(desired_action,
169  observation);
170  safe_action_->append(safe_action);
171  observation_->append(observation);
172 
173  // safe_action_->append(constrain_action();
174 
175  apply_action((*safe_action_)[t]);
176 
177  if (t >= 1)
178  {
179  check_timing(observation_->timestamp_ms(t) -
180  observation_->timestamp_ms(t - 1));
181  }
182  }
183 
184  // TODO: we have to make sure this loop exits properly at destruction time!!
185  }

188  void check_timing(const double &actual_step_duration_ms)
189  {
190  double max_step_duration_ms =
191  expected_step_duration_ms_ * step_duration_tolerance_ratio_;
192  double min_step_duration_ms =
193  expected_step_duration_ms_ / step_duration_tolerance_ratio_;
194 
195  if (is_realtime_ &&
196  (actual_step_duration_ms > max_step_duration_ms ||
197  actual_step_duration_ms < min_step_duration_ms))
198  {
199  std::ostringstream oss;
200  oss << "control loop did not run at expected rate "
201  "did you provide actions fast enough? "
202  << std::endl
203  << "expected step duration: "
204  << expected_step_duration_ms_ << std::endl
205  << "actual step duration: "
206  << actual_step_duration_ms << std::endl
207  << "tolearance ratio: "
208  << step_duration_tolerance_ratio_ << std::endl;
209  throw std::runtime_error(oss.str());
210  }

214  virtual Action compute_safe_action(const Action &desired_action,

216  {
217  // Vector kd(0.04, 0.08, 0.02);
218 
219  Vector kd(0.08, 0.08, 0.04);
220  double max_torque = 0.36;
221  Action safe_action = mct::clamp(desired_action,
222  -max_torque, max_torque);
223  safe_action = safe_action - kd.cwiseProduct(observation.velocity);
224  safe_action = mct::clamp(safe_action,
225  -max_torque, max_torque);
226  return safe_action;
227  }
