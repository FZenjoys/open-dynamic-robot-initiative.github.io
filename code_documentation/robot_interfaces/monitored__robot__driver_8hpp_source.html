177  void loop()
178  {
179  real_time_tools::set_cpu_dma_latency(0);
180 
181  // wait for the first data
182  while (!is_shutdown_ &&
183  !action_start_logger_.wait_for_timeindex(0, 0.1))
184  {
185  }
186 
187  // loop until shutdown and monitor action timing
188  for (size_t t = 0; !is_shutdown_; t++)
189  {
190  bool action_has_ended_on_time =
191  action_end_logger_.wait_for_timeindex(t,
192  max_action_duration_s_);
193  if (!action_has_ended_on_time)
194  {
195  error_message_.set(
196  "Action did not end on time, shutting down.");
197  shutdown();
198  return;
199  }
200 
201  bool action_has_started_on_time =
202  action_start_logger_.wait_for_timeindex(
203  t + 1, max_inter_action_duration_s_);
204  if (!action_has_started_on_time)
205  {
206  error_message_.set(
207  "Action did not start on time, shutting down.");
208  shutdown();
209  return;
210  }
211  }
212  }
