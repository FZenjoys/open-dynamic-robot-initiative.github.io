85  if (!processed_action.position.array().isNaN().all())

93  default_position_control_kd, desired_action.position_kd);
94 
95  typename Types::Vector position_error =
96  processed_action.position - latest_observation.position;
97 
98  // simple PD controller
99  typename Types::Vector position_control_torque =
100  processed_action.position_kp.cwiseProduct(position_error) -
101  processed_action.position_kd.cwiseProduct(
102  latest_observation.velocity);
103 
104  // position_control_torque contains NaN for joints where target
105  // position is set to NaN! Filter those out and set the torque to
106  // zero instead.
107  position_control_torque =
108  position_control_torque.array().isNaN().select(
109  0, position_control_torque);
110 
111  // Add result of position controller to the torque command
112  processed_action.torque += position_control_torque;
113  }
114 
115  // Safety Checks
116  // -------------
117  // limit to configured maximum torque
118  processed_action.torque =
119  mct::clamp(processed_action.torque, -max_torque_Nm, max_torque_Nm);
120  // velocity damping to prevent too fast movements
121  processed_action.torque -=
122  safety_kd.cwiseProduct(latest_observation.velocity);
123  // after applying checks, make sure we are still below the max. torque
124  processed_action.torque =
125  mct::clamp(processed_action.torque, -max_torque_Nm, max_torque_Nm);
126 
127  return processed_action;
