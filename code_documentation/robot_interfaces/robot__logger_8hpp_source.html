188  void append_robot_data_to_file()
189  {
190  output_file_.open(output_file_name_, std::ios_base::app);
191  output_file_.precision(27);
192 
193  for (long int j = index_;
194  j < std::min(index_ + block_size_,
195  logger_data_->observation->newest_timeindex());
196  j++)
197  {
198  try
199  {
200  Action applied_action = (*logger_data_->applied_action)[j];
201  Action desired_action = (*logger_data_->desired_action)[j];
202  Observation observation = (*logger_data_->observation)[j];
203  Status status = (*logger_data_->status)[j];
204 
205  std::vector<std::vector<double>> status_data =
206  status.get_data();
207  std::vector<std::vector<double>> observation_data =
208  observation.get_data();
209 
210  std::vector<std::vector<double>> applied_action_data =
211  applied_action.get_data();
212 
213  std::vector<std::vector<double>> desired_action_data =
214  desired_action.get_data();
215 
216  output_file_ << logger_data_->observation->timestamp_s(j) << " "
217  << j << " ";
218 
219  append_field_data_to_file(status_data);
220  append_field_data_to_file(observation_data);
221  append_field_data_to_file(applied_action_data);
222  append_field_data_to_file(desired_action_data);
223 
224  output_file_ << std::endl;
225  }
226 
227  catch (const std::exception &e)
228  {
229  std::cout << "Trying to access index older than the "
230  "oldest! Skipping ahead."
231  << std::endl;
232  }
233  }
234 
235  output_file_.close();
236  }

266  void write()
267  {
268  append_header_to_file();
269 
270  while (!stop_was_called_ &&
271  !(logger_data_->desired_action->length() > 0))
272  {
273  real_time_tools::Timer::sleep_until_sec(0.1);
274  }
275 
276  index_ = logger_data_->observation->newest_timeindex();
277 
278  while (!stop_was_called_)
279  {
280  if (index_ + block_size_ <=
281  logger_data_->observation->newest_timeindex())
282  {
283 #ifdef VERBOSE
284  auto t1 = std::chrono::high_resolution_clock::now();
285 #endif
286 
287  append_robot_data_to_file();
288 
289  index_ += block_size_;
290 
291 #ifdef VERBOSE
292 
293  // to check whether the data being requested to be logged is in
294  // the buffer of the timeseries and inspect effect of delays
295  std::cout << "Index trying to access, oldest index in the "
296  "buffer: "
297  << j << ","
298  << logger_data_->observation->oldest_timeindex()
299  << std::endl;
300 
301  auto t2 = std::chrono::high_resolution_clock::now();
302  auto duration =
303  std::chrono::duration_cast<std::chrono::microseconds>(t2 -
304  t1)
305  .count();
306 
307  // to print the time taken for one block of data to be logged.
308  std::cout << "Time taken for one block of data to be logged: "
309  << duration << std::endl;
310 #endif
311  }
312  }
