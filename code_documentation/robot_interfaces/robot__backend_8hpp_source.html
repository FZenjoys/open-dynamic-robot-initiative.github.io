124  void loop()
125  {
126  timers_.resize(10);
127  real_time_tools::set_cpu_dma_latency(0);
128 
129  // wait until first desired_action was received
130  // ----------------------------
131  while (!destructor_was_called_ &&
132  !robot_data_->desired_action->wait_for_timeindex(0, 0.1))
133  {
134  }
135 
136  for (long int t = 0; !destructor_was_called_; t++)
137  {
138  // TODO: figure out latency stuff!! open /dev/cpu_dma_latency:
139  // Permission denied
140 
141  timers_[0].tac_tic();
142 
143  timers_[6].tic();
144  // get latest observation from robot and append it to robot_data_
145  // --------
146  Observation observation = robot_driver_.get_latest_observation();
147  timers_[6].tac();
148 
149  timers_[1].tic();
150  robot_data_->observation->append(observation);
151  // TODO: for some reason this sometimes takes more than 2 ms
152  // i think this may be due to a non-realtime thread blocking the
153  // timeseries. this is in fact an issue, we might have to duplicate
154  // all the timeseries and have a realtime thread writing back and
155  // forth
156  timers_[1].tac();
157 
158  timers_[2].tic();
159  // if the robot has a finite max_inter_action_duration_s (not NAN or
160  // infinite, meaning it requires receiving actions in fixed time
161  // intervals), but robot_data_ has not received yet the next action
162  // to apply, we optionally repeat the previous action.
163  Status status;
164  if (std::isfinite(
165  robot_driver_.get_max_inter_action_duration_s()) &&
166  robot_data_->desired_action->newest_timeindex() < t)
167  {
168  uint32_t action_repetitions =
169  robot_data_->status->newest_element().action_repetitions;
170 
171  if (action_repetitions < max_action_repetitions_)
172  {
173  robot_data_->desired_action->append(
174  robot_data_->desired_action->newest_element());
175  status.action_repetitions = action_repetitions + 1;
176  }
177  }
178  robot_data_->status->append(status);
179  timers_[2].tac();
180 
181  timers_[3].tic();
182  // TODO: this may wait forever
183  Action desired_action = (*robot_data_->desired_action)[t];
184  timers_[3].tac();
185  timers_[4].tic();
186  Action applied_action = robot_driver_.apply_action(desired_action);
187  timers_[4].tac();
188  timers_[5].tic();
189  robot_data_->applied_action->append(applied_action);
190  timers_[5].tac();
191 
192  // if (t % 5000 == 0) {
193  // for (size_t i = 0; i < 7; i++) {
194  // std::cout << i << " --------------------------------------"
195  // << std::endl;
196  // timers_[i].print_statistics();
197  // }
198  // }
199  }
200  }
