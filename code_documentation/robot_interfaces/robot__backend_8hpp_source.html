184  void loop()
185  {
186  real_time_tools::set_cpu_dma_latency(0);
187 
188  // wait until first desired_action was received
189  // ----------------------------
190  while (!has_shutdown_request() &&
191  !robot_data_->desired_action->wait_for_timeindex(0, 0.1))
192  {
193  }
194 
195  for (long int t = 0; !has_shutdown_request(); t++)
196  {
197  // TODO: figure out latency stuff!!
198 
199  timer_.start();
200 
201  // get latest observation from robot and append it to
202  // robot_data_
203  // --------
204  Observation observation = robot_driver_.get_latest_observation();
205  timer_.checkpoint("get observation");
206 
207  robot_data_->observation->append(observation);
208  // TODO: for some reason this sometimes takes more than 2 ms
209  // i think this may be due to a non-realtime thread blocking the
210  // timeseries. this is in fact an issue, we might have to
211  // duplicate all the timeseries and have a realtime thread
212  // writing back and forth
213  timer_.checkpoint("append observation");
214 
215  Status status;
216  // If real time mode is enabled the next action needs to be provided
217  // in time. If this is not the case, optionally repeat the previous
218  // action or raise an error.
219  if (real_time_mode_ &&
220  robot_data_->desired_action->newest_timeindex() < t)
221  {
222  uint32_t action_repetitions =
223  robot_data_->status->newest_element().action_repetitions;
224 
225  if (action_repetitions < max_action_repetitions_)
226  {
227  robot_data_->desired_action->append(
228  robot_data_->desired_action->newest_element());
229  status.action_repetitions = action_repetitions + 1;
230  }
231  else
232  {
233  // No action provided and number of allowed repetitions
234  // of the previous action is exceeded --> Error
235  status.error_status = Status::ErrorStatus::BACKEND_ERROR;
236  status.error_message =
237  "Next action was not provided in time";
238  }
239  }
240 
241  std::string driver_error_msg = robot_driver_.get_error();
242  if (!driver_error_msg.empty())
243  {
244  status.error_status = Status::ErrorStatus::DRIVER_ERROR;
245  status.error_message = driver_error_msg;
246  }
247 
248  robot_data_->status->append(status);
249 
250  // if there is an error, shut robot down and stop loop
251  if (status.error_status != Status::ErrorStatus::NO_ERROR)
252  {
253  std::cerr << "Error: " << status.error_message
254  << "\nRobot is shut down." << std::endl;
255  break;
256  }
257  timer_.checkpoint("status");
258 
259  // early exit if destructor has been called
260  while (!has_shutdown_request() &&
261  !robot_data_->desired_action->wait_for_timeindex(t, 0.1))
262  {
263  }
264  if (has_shutdown_request())
265  {
266  break;
267  }
268 
269  Action desired_action = (*robot_data_->desired_action)[t];
270  timer_.checkpoint("get action");
271 
272  Action applied_action = robot_driver_.apply_action(desired_action);
273  timer_.checkpoint("apply action");
274 
275  robot_data_->applied_action->append(applied_action);
276  timer_.checkpoint("append applied action");
277 
278  if (t % 5000 == 0 && t > 0)
279  {
280  timer_.print_statistics();
281  }
282  }
283 
284  robot_driver_.shutdown();
285  loop_is_running_ = false;
286  }
