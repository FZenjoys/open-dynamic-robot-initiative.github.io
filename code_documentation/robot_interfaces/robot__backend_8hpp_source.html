131  void loop()
132  {
133  real_time_tools::set_cpu_dma_latency(0);
134 
135  // wait until first desired_action was received
136  // ----------------------------
137  while (!destructor_was_called_ &&
138  !robot_data_->desired_action->wait_for_timeindex(0, 0.1))
139  {
140  }
141 
142  for (long int t = 0; !destructor_was_called_; t++)
143  {
144  // TODO: figure out latency stuff!!
145 
146  timer_.start();
147 
148  // get latest observation from robot and append it to
149  // robot_data_
150  // --------
151  Observation observation = robot_driver_.get_latest_observation();
152  timer_.checkpoint("get observation");
153 
154  robot_data_->observation->append(observation);
155  // TODO: for some reason this sometimes takes more than 2 ms
156  // i think this may be due to a non-realtime thread blocking the
157  // timeseries. this is in fact an issue, we might have to
158  // duplicate all the timeseries and have a realtime thread
159  // writing back and forth
160  timer_.checkpoint("append observation");
161 
162  // if the robot has a finite max_inter_action_duration_s (not
163  // NAN or infinite, meaning it requires receiving actions in
164  // fixed time intervals), but robot_data_ has not received yet
165  // the next action to apply, we optionally repeat the previous
166  // action.
167  Status status;
168  if (std::isfinite(
169  robot_driver_.get_max_inter_action_duration_s()) &&
170  robot_data_->desired_action->newest_timeindex() < t)
171  {
172  uint32_t action_repetitions =
173  robot_data_->status->newest_element().action_repetitions;
174 
175  if (action_repetitions < max_action_repetitions_)
176  {
177  robot_data_->desired_action->append(
178  robot_data_->desired_action->newest_element());
179  status.action_repetitions = action_repetitions + 1;
180  }
181  else
182  {
183  // No action provided and number of allowed repetitions
184  // of the previous action is exceeded --> Error
185  status.error_status = Status::ErrorStatus::BACKEND_ERROR;
186  status.error_message =
187  "Next action was not provided in time";
188  }
189  }
190 
191  std::string driver_error_msg = robot_driver_.get_error();
192  if (!driver_error_msg.empty())
193  {
194  status.error_status = Status::ErrorStatus::DRIVER_ERROR;
195  status.error_message = driver_error_msg;
196  }
197 
198  robot_data_->status->append(status);
199 
200  // if there is an error, shut robot down and stop loop
201  if (status.error_status != Status::ErrorStatus::NO_ERROR)
202  {
203  std::cerr << "Error: " << status.error_message
204  << "\nRobot is shut down." << std::endl;
205  robot_driver_.shutdown();
206  return;
207  }
208  timer_.checkpoint("status");
209 
210  // early exit if destructor has been called
211  while (!robot_data_->desired_action->wait_for_timeindex(t, 0.1))
212  {
213  if (destructor_was_called_)
214  {
215  // TODO should shut down robot?
216  return;
217  }
218  }
219  Action desired_action = (*robot_data_->desired_action)[t];
220  timer_.checkpoint("get action");
221 
222  Action applied_action = robot_driver_.apply_action(desired_action);
223  timer_.checkpoint("apply action");
224 
225  robot_data_->applied_action->append(applied_action);
226  timer_.checkpoint("append applied action");
227 
228  if (t % 5000 == 0 && t > 0)
229  {
230  timer_.print_statistics();
231  }
232  }
233  }
