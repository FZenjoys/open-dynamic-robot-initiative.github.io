76  {
77  // check error state. do not allow appending actions if there is an
78  // error
79  if (robot_data_->status->length() > 0)
80  {
81  const Status status = robot_data_->status->newest_element();
82  switch (status.error_status)
83  {
84  case Status::ErrorStatus::NO_ERROR:
85  break;
86  case Status::ErrorStatus::DRIVER_ERROR:
87  throw std::runtime_error("Driver Error: " +
88  status.error_message);
89  case Status::ErrorStatus::BACKEND_ERROR:
90  throw std::runtime_error("Backend Error: " +
91  status.error_message);
92  default:
93  throw std::runtime_error("Unknown Error: " +
94  status.error_message);
95  }
96  }
97 
98  // since the timeseries has a finite memory, we need to make sure that
99  // by appending new actions we do not forget about actions which have
100  // not been applied yet
101  if (robot_data_->desired_action->length() ==
102  robot_data_->desired_action->max_length() &&
103  robot_data_->desired_action->oldest_timeindex() == // FIXME >=
104  get_current_timeindex())
105  {
106  std::cout

109  << std::endl;
110  wait_until_timeindex(
111  robot_data_->desired_action->oldest_timeindex() + 1);
112  }
113 
114  robot_data_->desired_action->append(desired_action);
